128void
129WorkDispatcher<I, O, Of, Ip, Op>::validate() const

140template <typename I, typename O, typename Of, typename Ip, typename Op>
141Of
142WorkDispatcher<I, O, Of, Ip, Op>::run(WorkGenerator<Ip> & generator,
143 WorkScheduler & scheduler) const
144{
145 validate();
146
147 torch::Device device = torch::kCPU;
148 std::size_t n = 0;
149 std::vector<Op> results;
150 while (generator.has_more())
151 {
152 // Wait until the worker is available
153 while (!scheduler.schedule_work(device, n))
154 // TODO: In theory this could hang if the worker pool is overloaded and no work is completed.
155 // We may want to add a timeout here.
156 continue;
157
158 neml_assert(n > 0, "Scheduler returned a batch size of ", n);
159
160 // Generate and preprocess work
161 auto && [m, work] = generator.next(n);
162 if (_preprocess)
163 work = _preprocess(std::move(work), device);
164
165 // Dispatch
166 auto result = _dispatch(std::move(work), device);
167
168 // Tell the scheduler that we have dispatched m batches. Since there is no asynchronous
169 // execution, we also immediately tell the scheduler that we have completed m batches
170 scheduler.dispatched_work(device, m);
171 scheduler.completed_work(device, m);
172
173 // Postprocess
174 if (_postprocess)
175 result = _postprocess(std::move(result));
176
177 results.push_back(result);
178 }
179
180 if (_reduce)
181 return _reduce(std::move(results));
182
183 if constexpr (std::is_same<Of, std::vector<Op>>::value)
184 return results;
185
186 throw NEMLException("Internal error: unreachable code");
187}

189template <typename I, typename O, typename Of, typename Ip, typename Op>
190Of
191WorkDispatcher<I, O, Of, Ip, Op>::run_async(WorkGenerator<Ip> & generator,
192 WorkScheduler & scheduler) const
193{
194 validate();
195
196 torch::Device device = torch::kCPU;
197 std::size_t n = 0;
198 using FutureResult = std::tuple<std::size_t, torch::Device, std::size_t, std::future<O>>;
199 std::vector<FutureResult> future_results;
200 std::vector<Op> results;
201 while (generator.has_more())
202 {
203 // Wait until the worker is available
204 while (!scheduler.schedule_work(device, n))
205 {
206 // Check if any of the dispatched work has completed
207 for (auto it = future_results.begin(); it != future_results.end();)
208 {
209 auto && [idx, device, m, future] = *it;
210 if (future.wait_for(std::chrono::seconds(0)) == std::future_status::ready)
211 {
212 auto result = future.get();
213 if (_postprocess)
214 result = _postprocess(std::move(result));
215 results[idx] = result;
216 scheduler.completed_work(device, m);
217 it = future_results.erase(it);
218 }
219 else
220 ++it;
221 }
222 }
223
224 neml_assert(n > 0, "Scheduler returned a batch size of ", n);
225
226 // Generate and preprocess work
227 auto && [m, work] = generator.next(n);
228 if (_preprocess)
229 work = _preprocess(std::move(work), device);
230
231 // Dispatch
232 auto result = std::async(std::launch::async, _dispatch, std::move(work), device);
233 future_results.emplace_back(results.size(), device, m, std::move(result));
234 results.resize(results.size() + 1); // Reserve space for the result
235
236 // Tell the scheduler that we have dispatched m batches.
237 scheduler.dispatched_work(device, m);
238 }
239
240 // Wait for all the dispatched work to complete
241 for (auto && [idx, device, m, future] : future_results)
242 {
243 future.wait();
244 auto result = future.get();
245 if (_postprocess)
246 result = _postprocess(std::move(result));
247 results[idx] = result;
248 scheduler.completed_work(device, m);
249 }
250
251 if (_reduce)
252 return _reduce(std::move(results));
253
254 if constexpr (std::is_same<Of, std::vector<Op>>::value)
255 return results;
256
257 throw NEMLException("Internal error: unreachable code");
258}

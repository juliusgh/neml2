190 template <typename T, typename = typename std::enable_if_t<std::is_base_of_v<Model, T>>>
191 T & register_model(const std::string & name, bool nonlinear = false, bool merge_input = true)
192 {
193 neml_assert(name != this->name(),
194 "Model named '",
195 this->name(),
196 "' is trying to register itself as a sub-model. This is not allowed.");
197
198 OptionSet extra_opts;
199 extra_opts.set<NEML2Object *>("_host") = host();
200 extra_opts.set<bool>("_nonlinear_system") = nonlinear;
201
202 auto model = Factory::get_object_ptr<Model>("Models", name, extra_opts);
203
204 if (merge_input)
205 for (auto && [name, var] : model->input_variables())
206 clone_input_variable(var);
207
208 _registered_models.push_back(model.get());
209 return *(std::dynamic_pointer_cast<T>(model));
210 }
